自動微分(Automatic Differentiation)¶

import torch

# 設定 x 參與自動微分
x = torch.tensor(4.0, requires_grad=True) 

y = x ** 2            # y = x^2

print(y)
print(y.grad_fn)      # y 梯度函數
y.backward()          # 反向傳導
print(x.grad)         # 取得梯度, dy/dx = 2x, 2x 在 x = 4 的取值 = 8

多次梯度下降¶

x = torch.tensor(5.0, requires_grad=True)
y = x ** 3            # y = x^3

y.backward(retain_graph=True) # 梯度下降
print(f'一次梯度下降={x.grad}') 

y.backward(retain_graph=True) # 梯度下降
print(f'二次梯度下降={x.grad}') 

y.backward() # 梯度下降
print(f'三次梯度下降={x.grad}')

x = torch.tensor(5.0, requires_grad=True)

z.backward() # 梯度下降

梯度下降法(Gradient Descent)¶

# 載入套件
import numpy as np
import matplotlib.pyplot as plt

# 目標函數(損失函數):y=x^2
def func(x): return x ** 2

# 目標函數的一階導數:dy/dx=2*x
def dfunc(x): return 2 * x

# 超參數(Hyperparameters)
x_start = 5     # 起始權重
epochs = 15     # 執行週期數 
lr = 0.3        # 學習率 

# 梯度下降法 
def GD(x_start, df, epochs, lr):    
    xs = np.zeros(epochs+1)    
    x = x_start    
    xs[0] = x    
    for i in range(epochs):         
        dx = df(x)        
        # 更新 x_new = x — learning_rate * gradient        
        x += - dx * lr         
        xs[i+1] = x    
    return xs

# *** Function 可以直接當參數傳遞 ***
w = GD(x_start, dfunc, epochs, lr=lr) 
print (np.around(w, 2))

t = np.arange(-6.0, 6.0, 0.01)
plt.plot(t, func(t), c='b')
plt.plot(w, func(w), c='r', marker ='o', markersize=5)    

# 設定中文字型
plt.rcParams['font.sans-serif'] = ['Microsoft JhengHei'] # 正黑體 
plt.rcParams['axes.unicode_minus'] = False # 矯正負號

plt.title('梯度下降法', fontsize=20)
plt.xlabel('X', fontsize=20)
plt.ylabel('損失函數', fontsize=20)
plt.show()

使用 PyTorch¶

# 載入套件
import torch
import numpy as np
import matplotlib.pyplot as plt

# 目標函數(損失函數):y=x^2
def func(x): return x ** 2

# 自動微分
def dfunc(x): 
    x = torch.tensor(float(x), requires_grad=True)
    y = x ** 2 # 目標函數(損失函數)
    y.backward()
    return x.grad

# 超參數(Hyperparameters)
x_start = 5     # 起始權重
epochs = 15     # 執行週期數 
lr = 0.3        # 學習率 

# 梯度下降法

    xs = np.zeros(epochs+1)    
    x = x_start    
    xs[0] = x    
    for i in range(epochs):         
        dx = df(x)        
        # x更新 x_new = x — learning_rate * gradient        
        x += - dx * lr         
        xs[i+1] = x    
    return xs

# *** Function 可以直接當參數傳遞 ***
w = GD(x_start, dfunc, epochs, lr=lr) 
print (np.around(w, 2))

t = np.arange(-6.0, 6.0, 0.01)

plt.plot(w, func(w), c='r', marker ='o', markersize=5)

使用 TensorFlow¶

# 載入套件
import tensorflow as tf
import numpy as np
import matplotlib.pyplot as plt

# 目標函數(損失函數):y=x^2
def func(x): return x ** 2

# 自動微分
def dfunc(x_value): 
    x = tf.Variable(x_value, dtype=tf.float32) # 宣告 TensorFlow 變數(Variable)
    with tf.GradientTape() as g: # 自動微分
        y = x ** 2                # y = x^2
    dy_dx = g.gradient(y, x)     # 取得梯度
    return dy_dx.numpy()         # 轉成 NumPy array

# 超參數(Hyperparameters)
x_start = 5     # 起始權重
epochs = 15     # 執行週期數 
lr = 0.3        # 學習率 

# 梯度下降法 
def GD(x_start, df, epochs, lr):    
    xs = np.zeros(epochs+1)    
    x = x_start    
    xs[0] = x    
    for i in range(epochs):         
        dx = df(x)        
        # x更新 x_new = x — learning_rate * gradient        
        x += - dx * lr         
        xs[i+1] = x    
    return xs

# *** Function 可以直接當參數傳遞 ***
w = GD(x_start, dfunc, epochs, lr=lr) 
print (np.around(w, 2))

t = np.arange(-6.0, 6.0, 0.01)
plt.plot(t, func(t), c='b')
plt.plot(w, func(w), c='r', marker ='o', markersize=5)    

# 設定中文字型
plt.rcParams['font.sans-serif'] = ['Microsoft JhengHei'] # 正黑體
